ðŸ§  SQL Queries Used

1. Total Revenue

SELECT ROUND(SUM(total), 2) AS total_revenue FROM invoices;
2. Revenue by Country

SELECT billing_country, ROUND(SUM(total), 2) AS revenue
FROM invoices GROUP BY billing_country ORDER BY revenue DESC;
3. Top 10 Customers

SELECT c.first_name || ' ' || c.last_name AS customer_name,
ROUND(SUM(i.total), 2) AS total_spent
FROM customers c JOIN invoices i ON c.customer_id = i.customer_id
GROUP BY customer_name ORDER BY total_spent DESC LIMIT 10;
4. Most Popular Genres

SELECT g.name AS genre, COUNT(ii.invoice_line_id) AS tracks_sold
FROM invoice_items ii
JOIN tracks t ON ii.track_id = t.track_id
JOIN genres g ON t.genre_id = g.genre_id
GROUP BY genre ORDER BY tracks_sold DESC;
